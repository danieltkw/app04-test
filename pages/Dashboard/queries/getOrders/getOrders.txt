-- Dynamic query using Appsmith's clientId (which is user_id in your case)
SELECT 
  o.*, 
  u.user_id AS client_id,
  o.shipping_address, 
  o.total, 
  o.delivery_status, 
  o.created
FROM db_orders o
INNER JOIN db_users u ON o.user_id = u.user_id
WHERE u.user_id = {{appsmith.store.clientId}};
